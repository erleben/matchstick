FILE(GLOB TetGen_HEADERS *.h )
FILE(GLOB TetGen_SOURCES *.cxx )

ADD_LIBRARY(tetgen
	    ${TetGen_SOURCES}
	    ${TetGen_HEADERS}
            )

SET(TetGen_DEF  "-D\"TETLIBRARY\"")

SET_TARGET_PROPERTIES(tetgen PROPERTIES COMPILE_FLAGS ${TetGen_DEF})
SET_TARGET_PROPERTIES(tetgen PROPERTIES DEBUG_POSTFIX "D" )

SET_TARGET_PROPERTIES(tetgen PROPERTIES ARCHIVE_OUTPUT_DIRECTORY                "${PROJECT_SOURCE_DIR}/lib/")
SET_TARGET_PROPERTIES(tetgen PROPERTIES ARCHIVE_OUTPUT_DIRECTORY_DEBUG          "${PROJECT_SOURCE_DIR}/lib/")
SET_TARGET_PROPERTIES(tetgen PROPERTIES ARCHIVE_OUTPUT_DIRECTORY_RELEASE        "${PROJECT_SOURCE_DIR}/lib/")
SET_TARGET_PROPERTIES(tetgen PROPERTIES ARCHIVE_OUTPUT_DIRECTORY_MINSIZEREL     "${PROJECT_SOURCE_DIR}/lib/")
SET_TARGET_PROPERTIES(tetgen PROPERTIES ARCHIVE_OUTPUT_DIRECTORY_RELWITHDEBINFO "${PROJECT_SOURCE_DIR}/lib/")
