INCLUDE_DIRECTORIES(
  ${PROJECT_SOURCE_DIR}/PROX/3RDPARTY/QHULL/QHULL/include
  )

FILE(
  GLOB
  QHULL_SOURCES
  ${PROJECT_SOURCE_DIR}/PROX/3RDPARTY/QHULL/QHULL/src/*.c
  )

FILE(
  GLOB
  QHULL_HEADERS
  ${PROJECT_SOURCE_DIR}/PROX/3RDPARTY/QHULL/QHULL/include/*.h
  )

ADD_LIBRARY(
  qhull
  ${QHULL_SOURCES}
  ${QHULL_HEADERS}
  )

SET_TARGET_PROPERTIES(qhull PROPERTIES DEBUG_POSTFIX "D" )

SET_TARGET_PROPERTIES(qhull PROPERTIES ARCHIVE_OUTPUT_DIRECTORY                "${PROJECT_SOURCE_DIR}/lib/")
SET_TARGET_PROPERTIES(qhull PROPERTIES ARCHIVE_OUTPUT_DIRECTORY_DEBUG          "${PROJECT_SOURCE_DIR}/lib/")
SET_TARGET_PROPERTIES(qhull PROPERTIES ARCHIVE_OUTPUT_DIRECTORY_RELEASE        "${PROJECT_SOURCE_DIR}/lib/")
SET_TARGET_PROPERTIES(qhull PROPERTIES ARCHIVE_OUTPUT_DIRECTORY_MINSIZEREL     "${PROJECT_SOURCE_DIR}/lib/")
SET_TARGET_PROPERTIES(qhull PROPERTIES ARCHIVE_OUTPUT_DIRECTORY_RELWITHDEBINFO "${PROJECT_SOURCE_DIR}/lib/")

IF(CMAKE_GENERATOR MATCHES Xcode)		
  SET_TARGET_PROPERTIES(qhull PROPERTIES XCODE_ATTRIBUTE_WARNING_CFLAGS "-Wall")
ENDIF(CMAKE_GENERATOR MATCHES Xcode)
